\section{Giới thiệu \& Động lực}
%%=============================================================

\begin{frame}{LLM-as-a-Judge: Là gì và Tại sao?}
  \begin{columns}[T]
    \column{0.55\textwidth}
      \textbf{LLM-as-a-Judge là gì?}
      \begin{itemize}
        \item Sử dụng một LLM mạnh (GPT-4, Gemini…) làm \alert{bộ đánh giá tự động} cho đầu ra của các LLM khác
        \item Ứng dụng: chuẩn mực chatbot, bảng xếp hạng (AlpacaEval, MT-Bench), mô hình phần thưởng RLHF
      \end{itemize}

      \vspace{0.6em}
      \textbf{Tại sao hấp dẫn?}
      \begin{itemize}
        \item \good{$\sim$99\% rẻ hơn} so với chú thích thủ công
        \item \good{Gần tức thì} và mở rộng vô hạn
        \item Tương quan cao với con người được báo cáo ($>$80\%)
      \end{itemize}

    \column{0.42\textwidth}
      \begin{block}{Con người vs.\ LLM-as-a-Judge}
        \small
        \begin{tabular}{lcc}
          \toprule
          \textbf{Tiêu chí} & \textbf{Con người} & \textbf{LLM} \\
          \midrule
          Tốc độ       & Chậm   & \good{Nhanh}  \\
          Chi phí      & Cao    & \good{Thấp}   \\
          Mở rộng & Thấp & \good{Cao}  \\
          Nhất quán    & Thấp   & \good{Cao}    \\
          Độ tin cậy   & \good{Gold} & \bad{?}  \\
          \bottomrule
        \end{tabular}
      \end{block}
      \vspace{0.4em}
      \small\alert{Dù có lợi thế vận hành, \textbf{độ tin cậy} của các LLM đánh giá vẫn là câu hỏi mở.}
  \end{columns}
\end{frame}

% ----
\begin{frame}{Tại Sao Đây Là Vấn Đề?}
  \begin{columns}[T]
    \column{0.5\textwidth}
      \textbf{Các sai lệch đã biết của bộ đánh giá LLM}
      \begin{itemize}
        \item \alert{Position bias} — ưu tiên câu trả lời đầu tiên trong pairwise
        \item \alert{Verbosity bias} — ưu tiên câu trả lời dài hơn bất kể độ chính xác
        \item \alert{Self-preference bias} — ưu tiên đầu ra từ cùng gia đình mô hình
      \end{itemize}

      \vspace{0.5em}
      \textbf{Bẫy tương quan}
      \begin{itemize}
        \item Nghiên cứu trước đo tương quan Pearson/Spearman \emph{tổng thể}
        \item Tương quan cao có thể \textbf{ẩn giấu} các lỗi hợp chi tiết
        \item Không trả lời được: \emph{LLM có thực sự phát hiện các loại lỗi cụ thể không?}
      \end{itemize}

    \column{0.47\textwidth}
      \begin{alertblock}{Khoảng Trống Nghiên Cứu}
        Chưa có nghiên cứu nào thực hiện \textbf{đánh giá chi tiết theo từng loại lỗi} của các LLM đánh giá.\\[0.4em]
        Chúng ta cần một \emph{checklist có thể giải thích} để chẩn đoán \textbf{blind spot} trước khi tin tưởng LLM làm bộ đánh giá.
      \end{alertblock}

      \vspace{0.4em}
      \begin{block}{Hậu Quả}
        Nếu các bộ đánh giá có blind spot ẩn $\Rightarrow$ xếp hạng leaderboard \textbf{bị sai lệch} $\Rightarrow$ các mô hình tệm hơn được triển khai sản xuất.
      \end{block}
  \end{columns}
\end{frame}

\begin{frame}{Các Câu Hỏi Nghiên Cứu}
  \textbf{Câu hỏi nghiên cứu cốt lõi (RQ):} \emph{Các bộ đánh giá LLM hiện tại có thể phát hiện đáng tin cậy các lỗi chi tiết qua 4 năng lực cốt lõi không?}

  \vspace{0.6em}
  \begin{columns}[T]
    \column{0.48\textwidth}
      \begin{block}{RQ1 — Viết Dạng Dài (LF)}
        Bộ đánh giá có bắt được lỗi mạch lạc, lỗi ngữ pháp, chính tả, mâu thuẫn thứ tự thời gian không?\\
        \small\textit{Ví dụ: ``These strategies \textbf{is}'' thay vì ``are''}
      \end{block}

      \vspace{0.4em}
      \begin{block}{RQ2 — Độ Chính Xác Thực Tế (F)}
        Bộ đánh giá có phát hiện được việc thay thế thực thể, số sai, sự kiện ngược lại không?\\
        \small\textit{Ví dụ: ``Poland'' $\to$ ``London''}
      \end{block}

    \column{0.48\textwidth}
      \begin{block}{RQ3 — Tuân Thủ Hướng Dẫn (IF)}
        Bộ đánh giá có xác định vi phạm định dạng, làm thiếu/thừa, bỏ qua ràng buộc không?\\
        \small\textit{Ví dụ: 6 dòng thay vì 4 dòng yêu cầu}
      \end{block}

      \vspace{0.4em}
      \begin{block}{RQ4 — Lập Luận (R)}
        Bộ đánh giá có phát hiện được lỗi số học, công thức sai, đơn vị không đúng không?\\
        \small\textit{Ví dụ: ``$2+3=6$'' thay vì $5$}
      \end{block}
  \end{columns}
\end{frame}

% ----
\begin{frame}{Giới thiệu Framework \fbi{}}
  \begin{columns}[T]
    \column{0.50\textwidth}
      \textbf{\fbi{} = Tìm Điểm Mù trong các LLM Đánh Giá bằng Checklist Có Thể Giải Thích}

      \vspace{0.4em}
      \textbf{Logic cốt lõi (dựa trên perturbation):}
      \begin{enumerate}
        \item Bắt đầu từ \emph{câu trả lời chuẩn} $A_{\text{gold}}$ chất lượng cao
        \item Thêm lỗi \emph{có kiểm soát} $\rightarrow A_{\text{perturb}}$
        \item Nạp cho bộ đánh giá; quan sát sự thay đổi điểm số
      \end{enumerate}

      \vspace{0.3em}
      \begin{align*}
        \text{Bộ đánh giá tốt:} &\quad s(A_{\text{perturb}}) < s(A_{\text{gold}}) \\
        \text{Blind spot:}       &\quad s(A_{\text{perturb}}) \approx s(A_{\text{gold}})
      \end{align*}

      \vspace{0.2em}
      \small\textbf{Quy mô:} 4 năng lực · 22 loại lỗi · 2.400 tuple · 17 người chú thích · 3 paradigm · 5 LLM

    \column{0.47\textwidth}
      \centering
      \includegraphics[width=\linewidth]{images/figure1_fbi_framework.png}
      \\\vspace{0.2em}\tiny\textit{Hình 1: Pipeline của framework \fbi{}}
  \end{columns}
\end{frame}

\section{Xây Dựng Benchmark}
%%=============================================================

\begin{frame}{Pipeline Xây Dựng Benchmark}
  \begin{columns}[T]
    \column{0.55\textwidth}
      \textbf{Đầu vào: 500 prompt} từ 6 dataset đã thiết lập
      \begin{itemize}
        \item WizardLM, MT-Bench, UltraChat, LIMA, LLMBar, IFEval
      \end{itemize}

      \vspace{0.4em}
      \textbf{Pipeline ba giai đoạn:}
      \begin{enumerate}
        \item \textbf{Tạo câu trả lời vàng} — GPT-4-Turbo để tạo ra các câu trả lời chuẩn xác nhất cho từng prompt.
        \item \textbf{Tạo perturbation} — GPT-4-Turbo Tạo ra các biến thể lỗi dựa trên 22 danh mục lỗi cụ thể.
        \item \textbf{Kiểm tra bởi con người} — 17 sinh viên sau đại học kiểm tra tính hợp lệ của perturbation.
      \end{enumerate}

      \vspace{0.4em}
      \textbf{Đầu ra:} 2.400 tuple $(\textit{prompt},\, A_{\text{gold}},\, A_{\text{perturb}})$

    \column{0.42\textwidth}
      \begin{alertblock}{Nguyên Tắc Thiết Kế}
        Benchmark \textbf{không} đo trực tiếp chất lượng câu trả lời. Nó đo \textbf{khả năng phát hiện lỗi của bộ đánh giá} — một nhiệm vụ khác và khó hơn về cơ bản.
      \end{alertblock}

      \vspace{0.4em}
      \begin{block}{Nhãn Annotation}
        \small
        \begin{itemize}
          \item \textbf{Valid} — lỗi rõ ràng tồn tại
          \item \textbf{Invalid} — lỗi không tồn tại / sai loại
          \item \textbf{Score-Invariant} — thay đổi vô hại
          \item \textbf{Not Relevant} — prompt lạc đề
          \item \textbf{Not Sure}
        \end{itemize}
      \end{block}
  \end{columns}
\end{frame}

\begin{frame}{Các Danh Mục Cốt Lõi}
  \begin{columns}[T]

    \column{0.55\textwidth}
      \begin{block}{\scriptsize Viết Dạng Dài (LF)}
        \scriptsize Bài luận sáng tạo/phân tích.
        \textit{``Làm thế nào để cải thiện quản lý thời gian?''}
        — lỗi cần \textbf{đọc toàn bộ tài liệu}.
      \end{block}
      \vspace{0.20em}
      \begin{block}{\scriptsize Thực Tế (F)}
        \scriptsize Câu hỏi có thể kiểm chứng khách quan.
        \textit{``Chức năng của tụ điện là gì?''}
        — lỗi cần kiểm tra \textbf{kiến thức thế giới}.
      \end{block}
      \vspace{0.20em}
      \begin{block}{\scriptsize Tuân Thủ Hướng Dẫn (IF)}
        \scriptsize Tạo sinh có ràng buộc.
        \textit{``Viết bài thơ 4 dòng dùng: peace, sky, race, ground.''}
        — lỗi cần \textbf{kiểm tra ràng buộc}.
      \end{block}
      \vspace{0.20em}
      \begin{block}{\scriptsize Lập Luận (R)}
        \scriptsize Bài toán toán học/logic.
        \textit{``Gậy+bóng = \$1,10; gậy đắt hơn \$1. Bóng giá bao nhiêu?''}
        — lỗi cần kiểm tra \textbf{từng bước}.
      \end{block}
  \end{columns}
\end{frame}

\begin{frame}{22 Danh Mục Perturbation (Bảng~2)}
  \begin{columns}[T]
    \column{0.52\textwidth}
      \centering
      \includegraphics[width=\linewidth]{images/22_perturbation.png}
      \\\vspace{0.15em}\tiny\textit{Bảng 2: Tất cả 22 danh mục perturbation qua 4 loại nhiệm vụ}

    \column{0.45\textwidth}
      \small
      \textbf{Nguyên tắc thiết kế:}
      \begin{itemize}
        \item Mỗi perturbation chỉ nhắm một \textbf{chiều lỗi}
        \item Lỗi có thể \textbf{giải thích được} — diễn giải với checklist
        \item Tránh các thay đổi đồng thời gây nhiễu attribution
      \end{itemize}

      \vspace{0.5em}
      \textbf{Ví dụ nhanh:}
      \begin{itemize}
        \item \textit{Entity:} ``Poland'' $\to$ ``London''
        \item \textit{Tính toán:} $2+3=5$ $\to$ $2+3=6$
        \item \textit{Do More:} thơ 4 dòng $\to$ 6 dòng
        \item \textit{Chronology:} sắp xếp lại sự kiện lịch sử
      \end{itemize}

      \vspace{0.4em}
      \begin{alertblock}{}
        \small 22 danh mục bao phủ \textbf{các lỗi thường gặp nhất trong thực tế} của văn bản LLM.
      \end{alertblock}
  \end{columns}
\end{frame}

        \item Lỗi quá nhẹ hoặc quá rõ ràng
        \item Giải thích nhắc đến lỗi nhưng \textbf{văn bản không phản ánh}
      \end{enumerate}

      \textbf{Thiết lập:} 17 sinh viên cao học review toàn bộ 2,400 perturbation\\
      Dán nhãn mỗi perturbation:\\
      \quad\textbf{Valid} / Invalid / Score-Invariant /\\
      \quad Không liên quan / Không chắc\\[0.3em]
      Invalid $\Rightarrow$ tạo lại cho đến khi hợp lệ.

      \vspace{0.4em}
      \begin{alertblock}{Score-Invariant (SI): 516 trường hợp}
        \small Thay đổi chỉ paraphrase \textbf{không nên} giảm điểm. Dùng để phát hiện \textbf{bộ đánh giá quá khắt khe}.
      \end{alertblock}
  \end{columns}
\end{frame}

\section{Các Chiến Lược Đánh Giá}

\begin{frame}{Ba Paradigm Đánh Giá}
  \begin{columns}[T]
    \column{0.30\textwidth}
      \begin{block}{Paradigm 1\\Chấm Điểm Một Câu Trả Lời}
        \centering
        \vspace{0.3em}
        Instruction + $A_{\text{perturb}}$\\
        $\downarrow$\\
        Điểm số (1–3)\\
        \vspace{0.3em}
        \small 5 chiến lược prompting
      \end{block}

    \column{0.30\textwidth}
      \begin{block}{Paradigm 2\\So Sánh Cặp (Pairwise)}
        \centering
        \vspace{0.3em}
        Instruction + $A_{\text{gold}}$ vs.\ $A_{\text{perturb}}$\\
        $\downarrow$\\
        Phán quyết (đầu)\\
        \vspace{0.3em}
        \small 5 chiến lược prompting
      \end{block}

    \column{0.36\textwidth}
      \begin{block}{Paradigm 3\\Chấm Điểm Có Tài Liệu Tham Chiếu}
        \centering
        \vspace{0.3em}
        Instruction + $A_{\text{gold}}$ (tham chiếu) + $A_{\text{perturb}}$\\
        $\downarrow$\\
        Điểm số (1–3)\\
        \vspace{0.3em}
        \small 1 chiến lược (thông tin nhất)
      \end{block}
  \end{columns}

  \vspace{0.7em}
  \begin{columns}[c]
    \column{0.6\textwidth}
      \textbf{Metric:} \% perturbation \textbf{không được phát hiện} (điểm số không đổi)\\
      $\downarrow$ thấp hơn là tốt \quad|\quad SI$\uparrow$ cao hơn là tốt (không false positive)
    \column{0.35\textwidth}
      \small\textbf{5 LLM đánh giá:} GPT-4-Turbo, Gemini-1.5-Pro, Claude-3-Opus, LLaMA-3-70B-Instruct, Prometheus~2
  \end{columns}
\end{frame}

% ----
\begin{frame}{Chấm Điểm Một Câu Trả Lời — 5 Chiến Lược}
  \begin{center}
  \small
  \begin{tabular}{llp{7cm}}
    \toprule
    \textbf{Chiến lược} & \textbf{Đầu vào} & \textbf{Đầu ra} \\
    \midrule
    \textbf{Vanilla*}    & Instruction + Answer & Chỉ điểm số \\
    \textbf{Vanilla}     & Instruction + Answer & Giải thích $\to$ Điểm số \\
    \textbf{Rubric}      & Instruction + Answer + Rubric chấm & Giải thích $\to$ Điểm số \\
    \textbf{Axis}        & Instruction + Answer + Trục đánh giá & Giải thích $\to$ Điểm số \\
    \textbf{Axis+Rubric} & Instruction + Answer + Axis + Rubric & Giải thích $\to$ Điểm số \\
    \bottomrule
  \end{tabular}
  \end{center}

  \vspace{0.5em}
  \begin{columns}[T]
    \column{0.48\textwidth}
      \textbf{Giả thuyết:} hướng dẫn nhiều hơn $\Rightarrow$ phát hiện tốt hơn\\[0.3em]
      \textit{Axis} = chiều khớp với nhiệm vụ (ví dụ: ``factual accuracy'' cho nhiệm vụ F)\\
      \textit{Rubric} = tiêu chí chấm điểm chi tiết theo từng chiều
    \column{0.48\textwidth}
      \begin{alertblock}{Tinh báo trước}
        Thêm rubric/axis \textbf{làm giảm} hiệu suất phát hiện trong chấm điểm một câu trả lời — sẽ giải thích thêm trong Kết quả.
      \end{alertblock}
  \end{columns}
\end{frame}

% ----
\begin{frame}{So Sánh Cặp (Pairwise) — 5 Chiến Lược}
  \begin{center}
  \small
  \begin{tabular}{llp{6cm}}
    \toprule
    \textbf{Chiến lược} & \textbf{Đầu vào} & \textbf{Đầu ra} \\
    \midrule
    \textbf{Pairwise*}  & Instruction + $A_1$, $A_2$ & Chỉ phán quyết \\
    \textbf{Pairwise}   & Instruction + $A_1$, $A_2$ & Giải thích $\to$ Phán quyết \\
    \textbf{Rules}      & Instruction + $A_1$, $A_2$ + Quy tắc & Giải thích $\to$ Phán quyết \\
    \textbf{Axis}       & Instruction + $A_1$, $A_2$ + Axis & Giải thích $\to$ Phán quyết \\
    \textbf{Axis+Rules} & Instruction + $A_1$, $A_2$ + Axis + Quy tắc & Giải thích $\to$ Phán quyết \\
    \bottomrule
  \end{tabular}
  \end{center}

  \vspace{0.5em}
  \begin{columns}[T]
    \column{0.55\textwidth}
      \textbf{Kiểm soát position bias:}
      \begin{itemize}
        \item Chạy \textbf{hai lần}, đảo vị trí của $A_{\text{gold}} / A_{\text{perturb}}$
        \item Chỉ tính là phát hiện nếu bộ đánh giá nhất quán ưu tiên $A_{\text{gold}}$
      \end{itemize}
    \column{0.40\textwidth}
      \begin{block}{Điểm khác chính với một câu trả lời}
        Trong pairwise, chiến lược nâng cao \textbf{có ích} — bộ đánh giá dùng quy tắc như \emph{checklist so sánh}, không phải neo điểm cố định.
      \end{block}
  \end{columns}
\end{frame}
